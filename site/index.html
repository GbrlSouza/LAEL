 Exemplo de Código 

component CounterApp:
    props:
        start_value: int = 0

    data:
        count: int = start_value

    methods:
        def increment(self):
            self.count += 1

    template:
        """
        <div>
            <p>Count: {{ count }}</p>
            <button @click="increment">Increment</button>
        </div>
        """
